create table vendedor (codigo int, nome varchar(50) not null, endereco varchar (150) not null, comissao int not null, CONSTRAINT pk_vendedor 
PRIMARY KEY (codigo));

create table cliente (codigoCliente int PRIMARY KEY, nomeCliente varchar(50) not null, enderecoCliente varchar (150) not null, faturamento float 
not null, limite int, codVendedor int not null, constraint fk_cliente foreign key (codVendedor) references vendedor(codigo));

create table armazem (codigoArmazem int PRIMARY KEY, enderecoArmazem varchar(150) not null);

create table peca (codigoPeca int PRIMARY KEY, descricao varchar (100), preco float not null, quantidade int not null, numeroArmazem int not null, constraint fk_peca foreign key (numeroArmazem) references armazem(codigoArmazem));
